 87def retrack_cs2_sin_max_coherence(
 88    l1b_file: str = "",
 89    waveforms: Union[np.ndarray, None] = None,
 90    coherence: Union[np.ndarray, None] = None,
 91    retrack_smooth_wf: bool = False,
 92    plot_flag: bool = False,
 93    measurement_index: Union[int, None] = None,
 94    include_measurements_array: Union[List[bool], None] = None,
 95    ref_bin_ind_sin: int = 512,
 96    wf_oversampling_factor: int = 100,
 97    noise_sample_limit: int = 6,
 98    noise_threshold: float = 0.30,
 99    savitsky_golay_width: int = 9,
100    savitsky_golay_poly_order: int = 3,
101    le_id_threshold: float = 0.05,
102    le_dp_threshold: float = 0.20,
103    coherence_smoothing_width=9,
104) -> Tuple[
105    np.ndarray,
106    np.ndarray,
107    List[List[float]],
108    List[List[float]],
109    np.ndarray,
110    int,
111    List[List[int]],
112]:
113    """
114    % SUMMARY
115    %   Retracking of CS2 SIN waveforms using Max Coherence Retracker
116    %   Reference: CryoSat-2 Product Handbook Baseline D1.1, C2-LI-ACS-ESL-5319.
117    %   This is an adapted python port of Matlab original by M.McMillan for Leading Edge
118    %   Detection, and
119    %   the LMC retracker in Aublanc et al, 2021 (Ice Sheet Topography from a
120    %   New CryoSat-2 SARIn Processing Chain, and Assessment by Comparison to ICESat-2
121    %   over Antarctica)
122    % ------------------------------------------------------------------------
123
124    Args:
125        l1b_file (str) : file name of L1b file
126        waveforms (np.ndarray): instead of reading from L1b file, you can pass in a
127                                numpy.ndarray of shape (num_measurements, waveform_numbins).
128                                waveform_numbins is 128 (LRM) or 1024 (SIN)
129        coherence (np.ndarray): SIN only coherence waveform for use in maximum coherence
130                                retracking
131        retrack_smooth_wf (bool, def=False): specify whether to retrack raw (False) or
132                                  smoothed waveform (True)
133        plot_flag (bool, def=False): set to True to plot waveforms and retracking points
134                                    for testing/debugging purposes
135                                    (Note, blocks until each plot closed)
136        measurement_index (int, def=None): if not None, only retrack this measurement index
137                                           (from 0). Used for debugging
138        include_measurements_array (List[bool]): None or [array of boolean values of size equal to
139                                    number of waveforms]. if not None, only retrack waveforms
140                                    corresponding to True values in this array
141        ref_bin_ind_sin (int, def=512) : reference bin index, from CS2 Baseline-D User
142                                         Manual, p36;
143        wf_oversampling_factor (int) : waveform oversampling factor (default=100)
144        noise_sample_limit (int, def=6) : maximum bin used to compute noise statistics
145        noise_threshold (float,def=0.3) : if mean amplitude in noise bins exceeds threshold then
146                                          reject waveform
147        savitsky_golay_width (int, def=9): Savitsky Golay smoothing width
148        savitsky_golay_poly_order (int, def=3): Savitsky Golay polynomial order
149        le_id_threshold (float, def=0.05) : power must exceed thermal noise by this amount to be
150                                            identified as a leading edge
151        le_dp_threshold (float, def=0.2): define threshold on normalised amplitude change which is
152                                          required to be accepted as lead edge
153        coherence_smoothing_width (int, def-9): coherence boxcar average smoothing width
154
155
156    Returns:
157        Tuple: (dr_bin_mc, dr_meters_mc, leading_edge_start, leading_edge_stop,pwr_at_rtrk_point_mc,
158                n_retrack_mc_failed, retrack_flag)
159                dr_bin_mc (List[float]) : max coherence epoch relative to nominal tracking point
160                                          in bins
161                dr_meters_mc (List[float]) : max coherence epoch relative to nominal tracking point
162                                          in meters
163                leading_edge_start (List[List[float]]): leading edge start coordinates
164                                                    column 1 = bin  |  column 2 = normalised power
165                leading_edge_stop (List[List[float]]): leading edge stop coordinates
166                                                    column 1 = bin  |  column 2 = normalised power
167                pwr_at_rtrk_point_mc (List[float]): power in counts at retracking point
168                n_retrack_mc_failed (int): number of waveforms were retracking failed
169                retrack_flag (List[List[int]]): returned retracker flags for each waveform indicate
170                            how retracking failed | 6 x t |
171                                column 1 (index 0): 0 or 1 max amplitude is 0 so skippings
172                                                    or mean noise above a predefined threshold
173                                column 2 (index 1): 0 or 1 if no samples are sufficiently above the
174                                    noise floor
175                                column 3 (index 2): 0 or 1 if no peak identified
176                                column 4 (index 3): 0 or 1 if no leading edge found by end of
177                                waveform
178                                column 5 (index 4): 0 (currently unused)
179                                column 6 (index 5): 0 or 1 if  No retracking point retrieved
180
181    """
182
183    if l1b_file and waveforms:
184        raise ValueError("Must have either l1b_file or waveforms, not both as input to function")
185
186    # -------------------------
187    # Find if input file is LRM or SIN. Only SIN allowed
188    # -------------------------
189
190    if l1b_file:
191        if "SIR_SIN_1B" in str(l1b_file):
192            mode_str = "SIN"
193        elif "SIR_LRM_1B" in str(l1b_file):
194            raise ValueError("Must be a file containing SARin waveforms, not LRM")
195        else:
196            raise ValueError(f"L1b file name {l1b_file} must include SIR_SIN_1B")
197    elif waveforms is not None:
198        if waveforms.ndim == 2 and waveforms.shape[1] == 1024:
199            n_waveforms, waveform_size = np.shape(waveforms)
200            mode_str = "SIN"
201        else:
202            raise ValueError("waveforms size must be (,1024) for SIN")
203
204    # -------------------------
205    # Define system parameters
206    # -------------------------
207
208    speed_of_light = 299792458  # speed of light (m/s) from CS2 Baseline-D User Manual, p36.
209    bandwidth = 320000000  # chirp bandwidth used (Hz) from from CS2 Baseline-D User Manual, p36.
210
211    # compute size of range bin
212    rbin_size_sin = speed_of_light / (
213        4 * bandwidth
214    )  # meters; from CS2 Baseline-D User Manual, p37. (0.2342m for CS2)
215
216    #   define Savitsky-Golay smoothing parameters
217    sm_width = savitsky_golay_width
218    sm_polynomial_order = savitsky_golay_poly_order
219
220    # -------------------------
221    # define quality thresholds
222    # -------------------------
223
224    # specify how noise is computed
225    noise_definition = "min_power"
226
227    if l1b_file:
228        # Load netcdf L1 data file
229        nc = Dataset(l1b_file)
230
231        # Retrieve the waveforms
232        wfs = nc.variables["pwr_waveform_20_ku"][:].data
233        n_waveforms, waveform_size = np.shape(wfs)
234
235        # Retrieve the coherence for SIN waveforms
236        coherence = nc.variables["coherence_waveform_20_ku"][:].data
237    else:
238        wfs = waveforms
239        # coherence passed as function parameter
240
241    # Check if the include_measurements_array is of the correct length
242    if include_measurements_array is not None:
243        if len(include_measurements_array) != n_waveforms:
244            raise InvalidArraySizeError(
245                f"include_measurements_array size {len(include_measurements_array)}  \
246                    must be same dimensions as number of waveforms {n_waveforms}"
247            )
248
249    # preallocate output arrays (list for each waveform)
250    leading_edge_start = [[np.nan for _ in range(2)] for _ in range(n_waveforms)]
251    leading_edge_stop = [[np.nan for _ in range(2)] for _ in range(n_waveforms)]
252    retrack_point_mc = [[np.nan for _ in range(3)] for _ in range(n_waveforms)]
253    retrack_flag = [[0 for _ in range(6)] for _ in range(n_waveforms)]
254
255    # Process each waveform
256    for i, waveform in enumerate(wfs):
257        # Special case for debugging individual measurements
258        if measurement_index:
259            if i != measurement_index:
260                continue
261
262        # log.debug('retracking measurment: {} '.format(i))
263
264        # skip waveform if include_measurements_array[i] is set to False
265        if include_measurements_array is not None:
266            if not include_measurements_array[i]:
267                log.debug("skip waveform as include_measurements_array[i] is set to False")
268                continue
269
270        log.debug("retracking waveform %d of %d", i, n_waveforms)
271
272        # compute max amplitude
273        wf_max = np.max(waveform)
274
275        if wf_max == 0.0:
276            log.debug("wf_max is 0 so skipping")
277            # set flag
278            retrack_flag[i][0] = 1
279            continue
280
281        # normalise so that max amplitude is 1
282        wfnorm = waveform / wf_max
283
284        # ---------
285        # smooth waveform
286        # ---------
287
288        # Apply 1-d Savitsky-Golay filter, to smooth waveform
289        wfnorm_sm = savgol_filter(wfnorm, sm_width, sm_polynomial_order)
290
291        # switch end values from 0 to nan
292        wfnorm_sm[wfnorm_sm == 0] = np.nan  # COMMENT : sets any 0 to Nan, not just end points?
293
294        # ---------------------
295        # compute thermal noise
296        # ---------------------
297        if noise_definition == "first_bins":
298            # estimate noise based on first samples of waveform
299            wf_noise_mean = np.mean(wfnorm[0:noise_sample_limit])
300
301        elif noise_definition == "min_power":
302            # alternatively sort power values of unsmoothed waveform in ascending order
303            wfsort = sorted(wfnorm)
304
305            # estimate noise based on lowest 6 samples
306            wf_noise_mean = np.mean(wfsort[0:6])
307        else:
308            raise ValueError(f"noise definition {noise_definition} unsupported")
309
310        # ------------------------------------------------------------
311        # quality check 1 - if mean noise above a predefined threshold
312        # ------------------------------------------------------------
313
314        if (wf_noise_mean > noise_threshold) or np.isnan(wf_noise_mean):
315            # set flag
316            retrack_flag[i][0] = 1
317
318            log.debug("quality check 1 FAILED : mean noise above a predefined threshold")
319
320            # do not attempt retracking and leave as nan
321
322        else:  # continue with retracking
323            # ------------------------
324            # Over Sample the waveform
325            # ------------------------
326            wf_bin_num = np.linspace(0, waveform_size - 1, waveform_size)
327
328            # create oversampled waveform bin indices
329            wf_bin_numi = np.linspace(0, waveform_size - 1, waveform_size * wf_oversampling_factor)
330
331            # Oversample normalised waveform
332            wfi = np.interp(wf_bin_numi, wf_bin_num, wfnorm)
333
334            # Oversample smoothed normalised waveform
335            wfi_sm = np.interp(wf_bin_numi, wf_bin_num, wfnorm_sm)
336
337            # ------------------
338            # compute derivative
339            # ------------------
340
341            # compute first derivative of smoothed waveform using central difference which is
342            # calculated between i-1:i+1
343            d_wf_sm = np.gradient(wfi_sm, (1 / wf_oversampling_factor))
344
345            # ------------------------------------
346            # initiate leading edge identification
347            # ------------------------------------
348
349            # initiate parameters for iteratively finding leading edge with amplitude above
350            # predefined threshold
351            previous_le_ind = 0
352            le_dp = 0
353            count = 1
354            # -------------------------------------------------------------------------------
355            # loop through leading edges until minimum amplitude requirement met or the end of
356            # the waveform is reached
357            # -------------------------------------------------------------------------------
358
359            # searches for leading edge greater than the amplitude threshold
360            while le_dp < le_dp_threshold:
361                # Find the first index of the waveform wfi_sm that satifies the criteria:
362                #    > (wf_noise_mean + le_id_threshold)
363                #    d_wf_sm > 0
364                #    index > previous_le_ind
365
366                le_index = np.where((wfi_sm > (wf_noise_mean + le_id_threshold)) & (d_wf_sm > 0))[0]
367
368                if le_index.size > 0:
369                    le_index = le_index[le_index > (previous_le_ind + wf_oversampling_factor)]
370
371                # ----------------------------------------------------------------------
372                # quality check 2 - if no samples are sufficiently above the noise floor
373                # ----------------------------------------------------------------------
374
375                if le_index.size == 0:
376                    # set flag
377                    retrack_flag[i][1] = 1
378                    log.debug(
379                        "quality check 2 FAILED :no samples are sufficiently above the noise floor"
380                    )
381                    # exit search for leading edge
382                    break
383
384                # Select the first leading edge index found
385                le_index = le_index[0]
386
387                # ----------------------------------------------------------
388                # leading edge exists so find position and amplitude of peak
389                # ----------------------------------------------------------
390
391                # find where the gradient first becomes negative after the power threshold is
392                # exceeded
393                first_peak_indices = np.where(
394                    (d_wf_sm <= 0) & (wf_bin_numi > wf_bin_numi[le_index])
395                )[0]
396                # Select the first one
397                if first_peak_indices.size > 0:
398                    first_peak_ind = int(first_peak_indices[0])
399
400                    # calculate the amplitude of the peak above the identified start point of
401                    # the leading edge
402                    le_dp = wfi_sm[first_peak_ind] - wfi_sm[le_index]
403
404                    # update previously identified peak to the current one in case the while
405                    # loop continues
406                    previous_le_ind = first_peak_ind
407
408                    # if reached end of waveform
409                    if previous_le_ind > (wf_bin_numi.size - wf_oversampling_factor - 1):
410                        # set flag
411                        retrack_flag[i][3] = 1
412                        # exit search for leading edge
413                        break
414                else:
415                    # ---------------------------------------------------------------------
416                    # quality check 3 - if no waveform peak can be identified after the
417                    # leading edge starts
418                    # ---------------------------------------------------------------------
419                    # first_peak_ind array is empty so set flag
420                    retrack_flag[i][2] = 1
421                    log.debug(
422                        "quality check 3 FAILED :no waveform peak can be identified after \
423                            the leading edge starts"
424                    )
425                    # exit search for leading edge
426                    break
427
428                # update count of how many iterations through loop
429                count = count + 1
430
431            # ----------------------
432            # find retracking point
433            # ----------------------
434
435            # only compute retracking points if no flags set
436
437            if np.sum(retrack_flag[i]) > 0:
438                log.debug("Retracker flags set, so not continuing to find retracking point")
439            else:
440                # ----------------------------------------------------------------------------
441                # find Max Coherence retracking point for SIN waveforms
442                # ----------------------------------------------------------------------------
443
444                # Smooth the coherence waveform using a running average window
445                # Coherence waveform is not oversampled
446                if coherence is not None:
447                    coherence_sm = fastsmooth(coherence[i], coherence_smoothing_width)
448                else:
449                    raise ValueError("coherence is None instead of np.ndarray")
450                # Find indices of 50% up the leading edge to start search for max coherence
451                #  where WFnorm [ns...ne] > 0.5 * LEamp + LEmin_energy
452                if retrack_smooth_wf:
453                    top_of_le_indices = np.where((wfi_sm - wfi_sm[le_index]) > 0.5 * le_dp)[0]
454                else:
455                    top_of_le_indices = np.where((wfi - wfi[le_index]) > 0.5 * le_dp)[0]
456                # Restrict to between le_index and first_peak_ind
457                top_of_le_indices = np.unique(
458                    np.rint(
459                        wf_bin_numi[
460                            top_of_le_indices[
461                                (top_of_le_indices >= le_index)
462                                & (top_of_le_indices <= first_peak_ind)
463                            ]
464                        ]
465                    ).astype(int)
466                )
467
468                if len(top_of_le_indices) < 1:
469                    retrack_flag[i][5] = 1
470                else:
471                    mc_start_index = top_of_le_indices[0]
472                    mc_end_index = top_of_le_indices[-1]
473                    # find the index of maximum coherence in wf indices
474                    index_of_max_coherence = top_of_le_indices[
475                        np.argmax(coherence_sm[top_of_le_indices])
476                    ]
477
478                    if retrack_smooth_wf:
479                        retrack_point_mc[i][0] = index_of_max_coherence
480                        retrack_point_mc[i][1] = wfi_sm[
481                            index_of_max_coherence * wf_oversampling_factor
482                        ]
483                        retrack_point_mc[i][2] = (
484                            wfi_sm[index_of_max_coherence * wf_oversampling_factor] * wf_max
485                        )
486                    else:
487                        retrack_point_mc[i][0] = index_of_max_coherence
488                        retrack_point_mc[i][1] = waveform[index_of_max_coherence] / wf_max
489                        retrack_point_mc[i][2] = waveform[index_of_max_coherence]
490
491                    if retrack_point_mc[i][2] == 0:
492                        retrack_point_mc[i][0] = np.nan
493                        retrack_point_mc[i][1] = np.nan
494                        retrack_point_mc[i][2] = np.nan
495                        log.debug("zero power found at retracking point")
496                        retrack_flag[i][5] = 1
497
498                if plot_flag:
499                    # Plot echo with retracking points
500                    fig = plt.figure(figsize=(15, 6))
501                    this_gridspec = gridspec.GridSpec(1, 2, width_ratios=[2.5, 1])
502                    ax1 = plt.subplot(this_gridspec[0])
503                    ax2 = plt.subplot(this_gridspec[1])
504
505                    ax1.plot(
506                        wf_bin_numi,
507                        wfi,
508                        color="green",
509                        marker=".",
510                        label="oversampled waveform",
511                    )
512                    ax1.plot(
513                        wf_bin_numi,
514                        wfi_sm,
515                        color="blue",
516                        marker=".",
517                        label="oversampled smoothed waveform",

Arguments:
  • l1b_file (str) : file name of L1b file
  • waveforms (np.ndarray): instead of reading from L1b file, you can pass in a numpy.ndarray of shape (num_measurements, waveform_numbins). waveform_numbins is 128 (LRM) or 1024 (SIN)
  • coherence (np.ndarray): SIN only coherence waveform for use in maximum coherence retracking
  • retrack_smooth_wf (bool, def=False): specify whether to retrack raw (False) or smoothed waveform (True)
  • plot_flag (bool, def=False): set to True to plot waveforms and retracking points for testing/debugging purposes (Note, blocks until each plot closed)
  • measurement_index (int, def=None): if not None, only retrack this measurement index (from 0). Used for debugging
  • include_measurements_array (List[bool]): None or [array of boolean values of size equal to number of waveforms]. if not None, only retrack waveforms corresponding to True values in this array
  • ref_bin_ind_sin (int, def=512) : reference bin index, from CS2 Baseline-D User Manual, p36;
  • wf_oversampling_factor (int) : waveform oversampling factor (default=100)
  • noise_sample_limit (int, def=6) : maximum bin used to compute noise statistics
  • noise_threshold (float,def=0.3) : if mean amplitude in noise bins exceeds threshold then reject waveform
  • savitsky_golay_width (int, def=9): Savitsky Golay smoothing width
  • savitsky_golay_poly_order (int, def=3): Savitsky Golay polynomial order
  • le_id_threshold (float, def=0.05) : power must exceed thermal noise by this amount to be identified as a leading edge
  • le_dp_threshold (float, def=0.2): define threshold on normalised amplitude change which is required to be accepted as lead edge
  • coherence_smoothing_width (int, def-9): coherence boxcar average smoothing width
